```
class LawOrchestrator:
    async def process(self, user_input: str) -> Response:
        # Step 1: 意图识别
        intent = await self.recognize_intent(user_input)
        
        # Step 2: 路由分发（就是if/elif）
        if intent == "regulation":
            return await self.handle_regulation(user_input)
        elif intent == "case":
            return await self.handle_case(user_input)
        elif intent == "document":
            return await self.handle_document(user_input)
        else:
            return await self.handle_unknown(user_input)
        
        # Step 3: 审计日志
        await self.audit.log(intent, result)
        
        return result
```

```
# 上下文记忆（简单实现）
class ConversationMemory:
    """简单的会话记忆，不需要复杂向量库"""
    
    def __init__(self, max_turns: int = 10):
        self.history = []  # List[Turn]
        self.max_turns = max_turns
    
    def add(self, user_input: str, agent_output: str):
        self.history.append(Turn(user_input, agent_output))
        if len(self.history) > self.max_turns:
            self.history.pop(0)
    
    def get_context(self) -> str:
        """获取压缩后的上下文"""
        return "\n".join([
            f"用户: {t.user}\n律师: {t.agent}"
            for t in self.history[-3:]  # 只保留最近3轮
        ])
```

### 3.4 响应格式化

```python
class StandardResponse:
    """标准响应格式"""
    
    def __init__(
        self,
        conclusion: str,
        legal_basis: List[dict],
        references: List[dict],
        risk_level: str,
        confidence: str,
        disclaimer: str
    ):
        self.conclusion = conclusion
        self.legal_basis = legal_basis
        self.references = references
        self.risk_level = risk_level
        self.confidence = confidence
        self.disclaimer = disclaimer
    
    def to_markdown(self) -> str:
        return f"""
## 【结论】
{self.conclusion}

## 【依据】
{chr(10).join([f"{i+1}. {b['title']} 第{b['article']}条" for i, b in enumerate(self.legal_basis)])}

## 【相关参考】
{chr(10).join([f"- {r['title']}" for r in self.references])}

## 【风险提示】
{self.risk_level}

## 【置信度】
{self.confidence}

## 【免责声明】
{self.disclaimer}
"""
```
